setwd('O:\\Users\\jun.yan\\Desktop\\桑格云绘图下载示例')
library(ggplot2)
df <- read.csv('kegg_enrichment.txt',header=TRUE,sep='\t')
p1 <- ggplot(df,aes(x=RichFactor,y=Term))
p2 <- p1 + geom_point(aes(color=p_value,size=ProteinNumber))
p3 <- p2 + ylab('pathway')
p4 <- p3 + scale_color_gradient(high='red', low='green')+theme_bw()
